  \item Which of the following Python data structures is \textbf{immutable}?
  \begin{enumerate}[label=(\Alph*)]
    \item List
    \item Dictionary
    \item Tuple
    \item Set
  \end{enumerate}

  \item What is the output of the following code?
  \begin{verbatim}
    my_dict = {'a': 1, 'b': 2, 'a': 3}
    print(my_dict['a'])
  \end{verbatim}
  \begin{enumerate}[label=(\Alph*)]
    \item 1
    \item 2
    \item 3
    \item KeyError
  \end{enumerate}

  \item Which method is used to add an element to the \textbf{end} of a list in Python?
  \begin{enumerate}[label=(\Alph*)]
    \item \texttt{insert()}
    \item \texttt{add()}
    \item \texttt{append()}
    \item \texttt{extend()}
  \end{enumerate}

  \item What will be the output of the following code?
  \begin{verbatim}
    s = {1, 2, 3, 2, 1}
    print(len(s))
  \end{verbatim}
  \begin{enumerate}[label=(\Alph*)]
    \item 5
    \item 3
    \item 2
    \item Error
  \end{enumerate}

  \item Which of the following correctly creates an empty dictionary?
  \begin{enumerate}[label=(\Alph*)]
    \item \texttt{d = []}
    \item \texttt{d = \{\}}
    \item \texttt{d = set()}
    \item \texttt{d = dict[]}
  \end{enumerate}
